
cmake_minimum_required(VERSION 3.7.2)

project(serial-rpi3b C)

add_library(serial_rpi3b_lib_fence INTERFACE)
target_include_directories(serial_rpi3b_lib_fence INTERFACE include/)

add_library(serial_rpi3b_lib_shared_dma src/shared_dma.c)
target_include_directories(serial_rpi3b_lib_shared_dma PUBLIC include/)

add_library(serial_rpi3b_lib_shared_ringbuffer src/libsharedringbuffer/shared_ringbuffer.c)
target_include_directories(serial_rpi3b_lib_shared_ringbuffer PUBLIC src/libsharedringbuffer/include/)
target_link_libraries(
        serial_rpi3b_lib_shared_ringbuffer
        lib_sel4cp_rpi3b
        serial_rpi3b_lib_fence
)

add_library(serial_rpi3b_lib_bit src/bit.c)
target_include_directories(serial_rpi3b_lib_bit PUBLIC include/)

add_library(serial_rpi3b_lib_serial_parity src/serial_parity.c)
target_include_directories(serial_rpi3b_lib_serial_parity PUBLIC include/)

add_library(serial_rpi3b_lib_bcm_uart_regs src/bcm_uart_regs.c)
target_include_directories(serial_rpi3b_lib_bcm_uart_regs PUBLIC include/)
target_link_libraries(
        serial_rpi3b_lib_bcm_uart_regs
        lib_sel4cp_rpi3b
        serial_rpi3b_lib_serial_parity
        serial_rpi3b_lib_bit
)

add_library(serial_rpi3b_lib_bcm_uart src/bcm_uart.c)
target_include_directories(serial_rpi3b_lib_bcm_uart PUBLIC include/)
target_link_libraries(
        serial_rpi3b_lib_bcm_uart
        lib_sel4cp_rpi3b
        serial_rpi3b_lib_bcm_uart_regs
)

add_library(serial_rpi3b_lib_serial_driver src/serial_driver.c)
target_include_directories(serial_rpi3b_lib_serial_driver PUBLIC include/)
target_link_libraries(
        serial_rpi3b_lib_serial_driver
        lib_sel4cp_rpi3b
        serial_rpi3b_lib_bcm_uart
        serial_rpi3b_lib_shared_ringbuffer
)

add_library(serial_rpi3b_lib_serial_client src/serial_client.c)
target_include_directories(serial_rpi3b_lib_serial_client PUBLIC include/)
target_link_libraries(
        serial_rpi3b_lib_serial_client
        lib_sel4cp_rpi3b
        serial_rpi3b_lib_shared_ringbuffer
        serial_rpi3b_lib_shared_dma
)
